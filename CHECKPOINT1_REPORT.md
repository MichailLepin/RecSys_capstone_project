```python
import math
from tqdm import tqdm

SOURCE = "recipes_with_embeddings.json"
CHUNK_SIZE_MB = 20
OUT_DIR = "chunks"

os.makedirs(OUT_DIR, exist_ok=True)

with open(SOURCE, "r") as f:
    data = json.load(f)

total = len(data)
recipes_per_chunk = int((CHUNK_SIZE_MB * 1024) / (len(json.dumps(data[0], ensure_ascii=False).encode("utf-8")) / 1024))
num_chunks = math.ceil(total / recipes_per_chunk)

for i in tqdm(range(num_chunks)):
    start = i * recipes_per_chunk
    end = start + recipes_per_chunk
    chunk = data[start:end]
    
    out_path = os.path.join(OUT_DIR, f"part{i+1}.json")
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(chunk, f, ensure_ascii=False)
```
